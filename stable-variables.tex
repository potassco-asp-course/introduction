\begin{frame}{Stable models of programs with Variables}
  \bigskip
  \begin{itemize}
  \item
    Let $P$ be a normal logic program with variables
    \medskip
  \item<2->
    A set $X$ of (\alert{ground}) atoms is a \alert{stable model} of $P$,
    \par\smallskip
    if $X$ is a stable model of \alert{\ground{P}}
  \end{itemize}
\end{frame}
