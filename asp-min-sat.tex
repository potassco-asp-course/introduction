\begin{frame}{Closed world reasoning}{by example}
  \bigskip
  \begin{itemize}
  \item Alphabet $\{a,b\}$
    \medskip
  \item The \only<1>{fact}\only<2>{rule}
    \[
      \only<2>{\neg b\to} a
    \]
  \item[] has the
    \par\smallskip
    \begin{itemize}\normalsize
    \item \makebox[75pt][l]{models}         $\{a\}$\only<2>{, $\{b\}$}, $\{a,b\}$
      \smallskip
    \item \makebox[75pt][l]{minimal models} $\{a\}$\only<2>{, $\{b\}$}
      \smallskip
    \item \makebox[75pt][l]{stable models}  $\{a\}$
    \end{itemize}
  \end{itemize}
\end{frame}
