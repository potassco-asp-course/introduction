\begin{frame}{Closed world reasoning}{by example}
  \bigskip
  \begin{itemize}
  \item The rule
    \[
      \neg b\to a
    \]
  \item[] has the
    \par\smallskip
    \begin{itemize}\normalsize
    \item \makebox[75pt][l]{models}         $\{a\}$, $\{b\}$, $\{a,b\}$
      \smallskip
    \item \makebox[75pt][l]{minimal models} $\{a\}$, $\{b\}$
      \smallskip
    \item \makebox[75pt][l]{stable models}  $\{a\}$
    \end{itemize}
  \end{itemize}
\end{frame}
