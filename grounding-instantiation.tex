\begin{frame}{Ground instantiation}

  \begin{itemize}
  \item<2-> Let $\mathcal{T}$ be a set of \invisible<6-7>{(}variable-free\invisible<6-7>{)} \alert<2-3>{terms}
    \smallskip
    \only<6  >{\\ (also called                           \alert{Herbrand universe})}
    \begin{itemize}\normalsize
    \item<only@3-5> \structure{Examples} \ $42$, ``$\mathit{cou cou}$'', $\mathit{Zorro}$, $\mathit{grandfather}(\mathit{leon})$, $3+X$
    \end{itemize}
    \smallskip
  \item<2-> Let $\mathcal{A}$ be a set of \invisible<6-7>{(}variable-free\invisible<6-7>{)} \alert<2-3>{atoms} constructible from $\mathcal{T}$
    \smallskip
    \only<6-7>{\\ (also called \alt<7>{\alert{alphabet}}{\alert{Herbrand base}})}
    \begin{itemize}\normalsize
    \item<only@3-5> \structure{Examples} \ $q(42)$, $\mathit{married}(\alt<5>{X}{\mathit{grandfather}(\alt<4>{X}{\mathit{leon}})})$, $\mathit{prime}(3+X)$
    \end{itemize}
    \medskip
  \item<8-> A variable-free atom is also called \alert{ground}
    \medskip
  \item<9-> \structure{Ground instances} of a rule $r$ are obtained by
    replacing all variables in $r$ by elements from $\mathcal{T}$:
    \[
    \ground{r}=\{r\theta\mid\theta:\mathit{var}(r)\rightarrow \mathcal{T} \text{ and } \mathit{var}(r\theta)=\emptyset\}
    \]
    where $\mathit{var}(r)$ stands for the set of all variables occurring in $r$ and

    $\theta$ is a (ground) substitution
    \medskip
  \item<10-> \structure{Ground instantiation} of logic program $P$
    \[
    \ground{P}=\textstyle\bigcup_{r\in P}\ground{r}
    \]
  \end{itemize}
\end{frame}
